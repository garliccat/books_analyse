Xfit = xfit[:, np.newaxis]
yfit = model.predict(Xfit)
print('Минимальная цена книги = {:.1f}'.format(y.min()))
print('Если бы продавали на вес, то 1 гр. книг стоил бы = {:.1f} руб.'.format(model.coef_[0]))
